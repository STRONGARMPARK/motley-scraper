            const content_div = document.querySelector("div.tailwind-article-body");

                  "div[data-pitch-placement='foolcom_sidebar_top']"
                );

                  const homeless_sidebar_top_pitch = sidebar_top_pitch
                    .parentNode.removeChild(sidebar_top_pitch);
                  content_div.insertBefore(homeless_sidebar_top_pitch, seventh_paragraph.nextSibling);
                }
              }
            }

            const desktopify_sidebar_modules = () => {
              if (fifth_paragraph) {
                const sidebar_service_pitch = document.querySelector(
                  "div[data-pitch-placement='foolcom_sidebar_service']"
                );
                if (sidebar_service_pitch) {
                  const homeless_sidebar_service_pitch = sidebar_service_pitch
                    .parentNode.removeChild(sidebar_service_pitch);
                  document
                    .getElementById("sidebar-service-pitch-wrapper")
                    .appendChild(homeless_sidebar_service_pitch);
                }
              }

              if (seventh_paragraph) {
                const sidebar_top_pitch = document.querySelector(
                  "div[data-pitch-placement='foolcom_sidebar_top']"
                );
                if (sidebar_top_pitch) {
                  const homeless_sidebar_top_pitch = sidebar_top_pitch
                    .parentNode.removeChild(sidebar_top_pitch);
                  document
                    .getElementById("sidebar-top-pitch-wrapper")
                    .appendChild(homeless_sidebar_top_pitch);
                }
              }
            };

            const toggle_sidebar_modules = (is_mobile) => {
              if (is_mobile) {
                mobilize_sidebar_modules();
              } else {
                desktopify_sidebar_modules();
              }
            };

            const mobile_query = window.matchMedia("(max-width: 640px)");
            if (mobile_query.matches) {
              mobilize_sidebar_modules();
            }
            mobile_query.addEventListener("change", (e) => {
              toggle_sidebar_modules(e.matches);
            });

          })();
